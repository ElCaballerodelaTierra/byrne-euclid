\documentclass{book}
\usepackage{byrne}
\begin{document}
\defineNewPicture{
	pair A, B, C;
	pair D, E, F;
	pair G, H, I;
	numeric s;
	s := 2/3u;
	A := (0, 0);
	B := (s, 0);
	C := dir(45) scaled s;
	byAngleDefine(B, A, C, byred, SOLID_SECTOR);
	byLineDefine(A, B, byblack, SOLID_LINE, REGULAR_WIDTH);
	byLineDefine(A, C, byblack, SOLID_LINE, REGULAR_WIDTH);
	draw byNamedAngleResized(BAC);
	draw byNamedLineSeq(0)(AB, AC);
	label.lft(btex A etex, A);
	D := (0, -1.5s);
	E := D shifted (s, 0);
	F := D shifted (dir(45) scaled s);
	byAngleDefine(E, D, F, byblue, SOLID_SECTOR);
	byLineDefine(D, E, byblack, SOLID_LINE, REGULAR_WIDTH);
	byLineDefine(D, F, byblack, SOLID_LINE, REGULAR_WIDTH);
	draw byNamedAngleResized(EDF);
	draw byNamedLineSeq(0)(DE, DF);
	label.lft(btex B etex, D);
	G := (0, -3.0s);
	H := G shifted (s, 0);
	I := G shifted (dir(45) scaled s);
	byAngleDefine(H, G, I, byyellow, SOLID_SECTOR);
	byLineDefine(G, H, byblack, SOLID_LINE, REGULAR_WIDTH);
	byLineDefine(G, I, byblack, SOLID_LINE, REGULAR_WIDTH);
	draw byNamedAngleResized(HGI);
	draw byNamedLineSeq(0)(GH, GI);
	label.lft(btex C etex, G);
}
